        // Format dates beautifully for human reading

        function formatDate(dateStr) {

            if (!dateStr) return '';

            const options = { year: 'numeric', month: 'long', day: 'numeric' };

            return new Date(dateStr).toLocaleDateString('en-US', options);

        }


        // Open Modal to enter/edit book log

        function openLogModal(bookNumber) {

            const modal = document.getElementById('log-modal');

            const modalContainer = document.getElementById('modal-container');

            const deleteBtn = document.getElementById('delete-btn');

                document.getElementById('modal-author').value = existingBook.author || '';

                document.getElementById('modal-date').value = existingBook.date || getTodayDateString();

                document.getElementById('modal-reflection').value = existingBook.reflection || '';

            } else {

                document.getElementById('modal-author').value = '';

        // Interactive Rating Star Handler

        function setRating(ratingValue) {

            document.getElementById('modal-rating').value = ratingValue;

            const stars = document.querySelectorAll('.star');

            const ratingText = document.getElementById('rating-text');

            

            const textOptions = ["Rate it!", "Hmm, okay", "Nice book!", "Really liked it!", "Amazing story!", "All-time Favorite!"];

            ratingText.textContent = textOptions[ratingValue] || "Rate it!";


            stars.forEach((star, index) => {

                if (index < ratingValue) {

            });

        }


        // Save entry details to state & local storage

        function saveBookEntry(e) {

            e.preventDefault();

            const bookNum = document.getElementById('current-book-num').value;

            const title = document.getElementById('modal-title').value.trim();

            const author = document.getElementById('modal-author').value.trim();

            const date = document.getElementById('modal-date').value;

            const rating = parseInt(document.getElementById('modal-rating').value);

            const reflection = document.getElementById('modal-reflection').value.trim();


            if (!title) return;


            readingJourney[bookNum] = { title, author, date, rating, reflection };

            localStorage.setItem('reads_journey_data', JSON.stringify(readingJourney));

            

            closeModal();

            initializeGrid();

            updateStats();

            updateBadges();

            updateJournal();

        }


        // Delete logged book

        function deleteBookEntry() {

                localStorage.setItem('reads_journey_data', JSON.stringify(readingJourney));

                

                closeModal();

                initializeGrid();

                updateStats();

                updateBadges();

                updateJournal();

            }

        }


        // Calculate and display progress stats

        function updateStats() {

            const readCount = Object.keys(readingJourney).length;

            const percent = Math.min(100, Math.round((readCount / 100) * 100));


            document.getElementById('stat-count').textContent = readCount;

            document.getElementById('progress-text').textContent = `${readCount} / 100 Books`;

        }

        // Helper: escape input to prevent basic markup injection

        function escapeHTML(str) {

            return str

                .replace(/&/g, '&amp;')

                .replace(/</g, '&lt;')

                .replace(/>/g, '&gt;')

                .replace(/"/g, '&quot;')

                .replace(/'/g, '&#039;');

        }


        // Fetch Today's local date

        function getTodayDateString() {

            const today = new Date();

            const yyyy = today.getFullYear();

            let mm = today.getMonth() + 1;

            let dd = today.getDate();

            if (dd < 10) dd = '0' + dd;

            if (mm < 10) mm = '0' + mm;

            return yyyy + '-' + mm + '-' + dd;

        }


        // Storage load

        function loadFromLocalStorage() {

            try {

                const storedJourney = localStorage.getItem('reads_journey_data');

                const storedSettings = localStorage.getItem('reads_settings');

                

                if (storedJourney) readingJourney = JSON.parse(storedJourney);

                if (storedSettings) settings = JSON.parse(storedSettings);

            } catch (e) {

                console.error("Local storage corruption detected. Resetting to empty defaults.");

                readingJourney = {};

            }

        }
